### Question 5.
```C++
#include <iostream>
using namespace std;

int main() {
    
    int N = 3, mat[N][N], sum = 0;
    
    int n = 1;
    for (int i = 0; i < N; i++){
        for (int j = 0; j < N; j++){
            if (i == j){ // --> Note: i = rows, j = columns
                mat[i][j] = 0;
            }
            else{
                mat[i][j] = 1;
            }
            if (j == N - n){
                mat[i][j] = 0;
                n++;
            }
        }
    }
    n = 1;
    for (int i = 0; i < N; i++){
        for (int j = 0; j < N; j++){
            if (i != j and j != N - n){
                sum = sum + mat[i][j];
            }
            if (j == N - n){
                n++;
            }
        }
    }
    for (int i = 0; i < N; i++){
        for (int j = 0; j < N; j++){
        if (j == 0){
            cout << "[";
        }
        cout << mat[i][j];
        if (j == N - 1){
            cout << "]\n";
        }
        else{
            cout << ", ";
        }
        }
    }
    cout << "sum = " << sum;

    return 0;
}
```
