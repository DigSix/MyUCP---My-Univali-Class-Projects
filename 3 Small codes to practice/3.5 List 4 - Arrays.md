###Question 1.
```C++
#include <iostream>
using namespace std;

int main() {

    int N = 5, vet[N], n;
    
    for (int i = 0; i < N; i++){ // --> Note: Filing the vector "vet" with user inputs
        cout << "Please enter an integer number.\n";
        cin >> n;
        vet[i] = n;
    }
    for (int i = 0; i < N; i++){ // --> Note: Showing the vector "vet" with a visibility touch
        if (i == 0){
            cout << "[";
        }
        cout << vet[i];
        if (i == N - 1){
            cout << "]";
        }
        else{
            cout << ", ";
        }
    }
    
    return 0;
}
```
